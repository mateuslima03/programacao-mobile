import 'package:flutter/material.dart';
import 'package:google_fonts/google_fonts.dart';

void main() {
  runApp(CatalogoViagensApp());
}

class CatalogoViagensApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'Catálogo de Viagens',
      theme: ThemeData.dark().copyWith(
        useMaterial3: true,  // Ativa o Material 3
        scaffoldBackgroundColor: Colors.black,
        textTheme: GoogleFonts.robotoSlabTextTheme(
          ThemeData.dark().textTheme,
        ).copyWith(
          bodyLarge: TextStyle(color: Colors.white),  // Texto claro no modo escuro
          bodyMedium: TextStyle(color: Colors.white),  // Texto claro no modo escuro
          titleLarge: TextStyle(color: Colors.white),  // Títulos visíveis
        ),
        appBarTheme: AppBarTheme(
          backgroundColor: Colors.black,
          titleTextStyle: TextStyle(color: Colors.white),  // Título da AppBar
        ),
      ),
      debugShowCheckedModeBanner: false,
      home: HomePage(),
    );
  }
}

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text('Catálogo de Viagens'),

      body: ListView.builder(
        padding: const EdgeInsets.all(16),

                                  : Icons.star_border,
                              color: Colors.amber,
                              size: 20,
                            );
                          }),
                          SizedBox(width: 8),
                          Text(
                            destino['avaliacao'].toString(),
                            style: TextStyle(color: Colors.amberAccent),
                          ),
                        ],
                      ),
                      const SizedBox(height: 12),

                      // Botões de favoritar e compartilhar
                      Row(
                        mainAxisAlignment: MainAxisAlignment.end,
                        children: [
                          IconButton(
                            icon: Icon(
                              destino['favorito']
                                  ? Icons.favorite
                                  : Icons.favorite_border,
                              color: Colors.pinkAccent,
                            ),
                            tooltip: 'Favoritar',
                            onPressed: () {
                              // Alterna o status de favorito
                              destino['favorito'] = !destino['favorito'];
                            },
                          ),
                          IconButton(
                            icon: Icon(Icons.share, color: Colors.blueAccent),
                            tooltip: 'Compartilhar',
                            onPressed: () {

              ],

// Tela de detalhes do destino com animação Hero
class DetalhesDestino extends StatelessWidget {
  final Map<String, dynamic> destino;

  const DetalhesDestino({required this.destino});

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text(destino['titulo']),
        leading: BackButton(),
      ),
      body: ListView(
        children: [
          Hero(
            tag: destino['titulo'],  // Transição suave da imagem
            child: Image.network(
              destino['imagem'],
              height: 240,
              width: double.infinity,
              fit: BoxFit.cover,
            ),
          ),
          Padding(
            padding: const EdgeInsets.all(24),
            child: Column(
              crossAxisAlignment: CrossAxisAlignment.start,
              children: [
                // Título do destino
                Text(
                  destino['titulo'],
                  style: GoogleFonts.robotoSlab(
                    fontSize: 26,
                    fontWeight: FontWeight.bold,
                    color: Colors.white,  // Título no fundo escuro
                  ),
                ),
                const SizedBox(height: 12),

                // Avaliação do destino em estrelas
                Row(
                  children: [
                    ...List.generate(5, (i) {
                      return Icon(
                        i < destino['avaliacao'].floor()
                            ? Icons.star
                            : Icons.star_border,
                        color: Colors.amber,
                      );
                    }),
                    const SizedBox(width: 8),
                    Text(
                      destino['avaliacao'].toString(),
                      style: TextStyle(color: Colors.amber),
                    ),
                  ],
                ),
                const SizedBox(height: 16),

                // Descrição longa do destino
                Text(
                  destino['textoDetalhado'],  // Texto detalhado do destino
                  style: Theme.of(context).textTheme.bodyLarge,
                ),
              ],
            ),
          ),
        ],
      ),
    );
  }
}
